```typescript
export const leavers = sqliteTable("leavers", {
  id: text("id").primaryKey(),
  migrationId: text("migration_id")
    .notNull()
    .references(() => migrations.id, { onDelete: "cascade" }),
  name: text("name").notNull(),
  linkedinUrl: text("linkedin_url"),
  currentTitle: text("current_title"),
  currentCompany: text("current_company"),
  transitionDate: text("transition_date"),
  createdAt: integer("created_at", { mode: "timestamp" }).notNull(),
});

export const leaverPositions = sqliteTable(
  "leaver_positions",
  {
    id: text("id").primaryKey(),
    leaverId: text("leaver_id")
      .notNull()
      .references(() => leavers.id, { onDelete: "cascade" }),
    company: text("company").notNull(),
    title: text("title").notNull(),
    startDate: text("start_date"),
    endDate: text("end_date"),
    sortOrder: integer("sort_order").notNull().default(0),
  },
  (table) => [index("leaver_positions_leaverId_idx").on(table.leaverId)],
);
```

Key design decisions:
- FK to `migrations` (not `searches`) per research recommendation -- downstream modal opens by clicking a role in a company card, which maps 1:1 to a migration record
- `transitionDate` as text (not integer timestamp) -- LinkedIn dates are fuzzy ("2024-03", "Jan 2024"), not precise timestamps
- `startDate`/`endDate` as text for same reason
- `sortOrder` for explicit position ordering (SQLite doesn't guarantee insertion order)
- Cascade deletes: searches -> migrations -> leavers -> positions (full chain)
